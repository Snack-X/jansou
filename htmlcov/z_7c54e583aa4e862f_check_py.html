1"""Scoring a parsed game against the values its log recorded. 

2 

3Each win is rebuilt into an `AgariRecord` by the replay, scored with the 

4library, and checked against the log: the rounded fu and, decisively, the win 

5value -- the points the win moves before honba and deposits. Reproducing that 

6value confirms the han total, the fu, the limit tier, and the dealer and 

7self-draw multipliers all at once. A round that our scorer cannot score, or 

8that reaches a different value, is reported rather than raised. 

9""" 

10 

11from __future__ import annotations

13from dataclasses import dataclass 

14from typing import TYPE_CHECKING 

15 

16from jansou.io.paifu import Paifu, replay_round 

17from jansou.scoring.score import ScoringError, score 

18

22_CHIITOITSU_FU = 25

25@dataclass(frozen=True) 

26class Verdict: 

27 """The outcome of checking one win: passed, or why it failed. 

28 

29 Attributes: 

30 winner: The seat index of the player who won the checked hand. 

31 passed: Whether the win reproduced the value (and fu) the log recorded. 

32 detail: A human-readable reason when ``passed`` is false, empty otherwise. 

33 """ 

34 

35 winner: int 

36 passed: bool 

37 detail: str = ""

40def check_win(record: AgariRecord) -> Verdict: 

41 """Score one rebuilt win and compare it to the value the log recorded. 

42 

43 The win is scored with the library, and its win value -- the points moved 

44 before honba and deposits -- is compared to the recorded value; the rounded 

45 fu is compared too, allowing the log's fixed-fu conventions, but not for a 

46 yakuman, whose value does not depend on fu. A hand the scorer cannot score 

47 fails rather than raising. 

48 

49 Args: 

50 record: The rebuilt ``AgariRecord`` for one win: its hand, winning tile, 

51 scoring context, and the expected value and fu from the log. 

52 

53 Returns: 

54 A passing ``Verdict``, or a failing one whose ``detail`` names the 

55 mismatch (unscorable, value differs, or fu differs). 

56 """ 

57 try: 

58 result = score(record.hand, record.winning_tile, record.context) 

59 except ScoringError as error: 

60 return Verdict(record.winner, passed=False, detail=f"unscorable: {error}") 

61 value = result.payment.total - result.payment.honba - result.payment.sticks 

62 if record.expected_value is not None and value != record.expected_value: 

63 return Verdict( 

64 record.winner, 

65 passed=False, 

66 detail=f"value {value} != expected {record.expected_value}", 

67 ) 

68 # A yakuman's value ignores fu, and logs record it inconsistently there, so fu is only checked otherwise. 

69 check_fu = record.expected_fu is not None and not result.is_yakuman 

70 if check_fu and not _fu_matches(result.fu.total, record.expected_fu): 

71 return Verdict( 

72 record.winner, 

73 passed=False, 

74 detail=f"fu {result.fu.total} != expected {record.expected_fu}", 

75 ) 

76 return Verdict(record.winner, passed=True) 

77 

78 

79def _fu_matches(computed: int, expected: int) -> bool: 

80 """Whether computed fu agrees, allowing the log's fixed-fu conventions.""" 

81 # A kazoe yakuman reports 0 fu; chiitoitsu reports 25. (True yakuman skip fu in the caller.) 

82 return expected in (0, _CHIITOITSU_FU) or computed == expected

85def check_paifu(paifu: Paifu) -> list[Verdict]: 

86 """Check every win in a parsed game. 

87 

88 Each round is replayed into its wins, and each win is scored and compared to 

89 the log via ``check_win``. 

90 

91 Args: 

92 paifu: The parsed game, carrying its rounds, rules, and player count. 

93 

94 Returns: 

95 One ``Verdict`` per win, in round-then-win order. 

96 """ 

97 return [ 

98 check_win(record) 

99 for round_log in paifu.rounds 

100 for record in replay_round(round_log, paifu.rules, paifu.player_count) 

101 ]
