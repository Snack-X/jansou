43def parse_mpsz(text: str) -> list[Tile]: 

44 """Parse MPSZ notation into tiles. 

45 

46 Runs may appear in any order and need not be sorted. Whitespace delimits 

47 runs; a run's digits must be followed by their suit letter before any 

48 whitespace or end of input. 

49 

50 Args: 

51 text: MPSZ notation such as ``"123m 0p 55s 77z"``, where a suit letter 

52 follows its digits and ``0`` denotes a red five. 

53 

54 Returns: 

55 The tiles named, in the order they appear in the text. 

56 

57 Raises: 

58 NotationError: If digits are not followed by a suit letter, a suit 

59 letter has no preceding digits, or an unexpected character appears. 

60 """ 

61 tiles: list[Tile] = [] 

62 pending: list[tuple[int, str]] = [] # (position, digit) awaiting a suit letter 

63 for position, char in enumerate(text): 

64 if char.isspace(): 

65 if pending: 

66 raise NotationError(f"digits without a suit letter at position {pending[0][0]}") 

67 elif char.isdigit(): 

68 pending.append((position, char)) 

69 elif char in _SUIT_OF_LETTER or char == "z": 

70 if not pending: 

71 raise NotationError(f"suit letter {char!r} with no preceding digits at position {position}") 

72 tiles.extend(_mpsz_run(char, pending)) 

73 pending.clear() 

74 else: 

75 raise NotationError(f"unexpected character {char!r} at position {position}") 

76 if pending: 

77 raise NotationError(f"digits without a suit letter at position {pending[0][0]}") 

78 return tiles

81def _mpsz_run(letter: str, digits: list[tuple[int, str]]) -> list[Tile]: 

82 """The tiles of one digit run given its suit letter.""" 

83 if letter == "z": 

84 tiles = [] 

85 for position, digit in digits: 

86 if digit not in _HONOR_DIGITS: 

87 raise NotationError(f"honor index {digit!r} out of range 1-7 at position {position}") 

88 tiles.append(Tile(TileKind(TileKind.EAST + int(digit) - 1))) 

89 return tiles 

90 suit = _SUIT_OF_LETTER[letter] 

91 return [ 

92 Tile.suited(suit, 5, red=True) if digit == _RED_RANK_DIGIT else Tile.suited(suit, int(digit)) 

93 for _, digit in digits 

94 ]

97def dump_mpsz(tiles: Iterable[Tile]) -> str: 

98 """Serialize tiles to MPSZ in canonical order. 

99 

100 A red five is written as `0` immediately before the ordinary fives of 

101 its suit; an empty collection yields the empty string. 

102 

103 Args: 

104 tiles: The tiles to serialize, in any order. 

105 

106 Returns: 

107 The canonical MPSZ string (suits grouped and sorted), or ``""`` when 

108 given no tiles. 

109 """ 

110 runs: dict[str, list[str]] = {"m": [], "p": [], "s": [], "z": []} 

111 for tile in sorted(tiles): 

112 if tile.is_honor: 

113 runs["z"].append(str(tile.kind - TileKind.EAST + 1)) 

114 else: 

115 suit: Suit = tile.suit # type: ignore[assignment] 

116 runs[suit.value].append(_RED_RANK_DIGIT if tile.red else str(tile.rank)) 

117 return "".join(f"{''.join(digits)}{letter}" for letter, digits in runs.items() if digits)

123def parse_mjai(text: str) -> list[Tile]: 

124 """Parse whitespace-separated MJAI tokens into tiles. 

125 

126 Args: 

127 text: MJAI tokens separated by whitespace, e.g. ``"5mr P 3s"`` (``5mr`` 

128 is a red five, honors are letters). 

129 

130 Returns: 

131 The tiles named, in order. 

132 

133 Raises: 

134 NotationError: If any token is not a recognized MJAI tile. 

135 """ 

136 return [_mjai_tile(token) for token in text.split()] 

137 

138 

139def _mjai_tile(token: str) -> Tile: 

140 """The tile named by one MJAI token.""" 

141 honor = _MJAI_HONOR_OF_LETTER.get(token) 

142 if honor is not None: 

143 return Tile(honor) 

144 if len(token) == 2 and token[0] in _RANK_DIGITS and token[1] in _SUIT_OF_LETTER: 

145 return Tile.suited(_SUIT_OF_LETTER[token[1]], int(token[0])) 

146 if len(token) == 3 and token[0] == "5" and token[1] in _SUIT_OF_LETTER and token[2] == "r": 

147 return Tile.suited(_SUIT_OF_LETTER[token[1]], 5, red=True) 

148 raise NotationError(f"unrecognized MJAI token {token!r}")

163def _mjai_token(tile: Tile) -> str: 

164 """The MJAI token naming one tile.""" 

165 if tile.is_honor: 

166 return _MJAI_LETTER_OF_HONOR[tile.kind] 

167 suit: Suit = tile.suit # type: ignore[assignment] 

168 return f"{tile.rank}{suit.value}{'r' if tile.red else ''}"

174def parse_136(indices: Iterable[int]) -> list[Tile]: 

175 """Parse 136-tile integers into tiles. 

176 

177 Each index names one physical tile, so a duplicate index is rejected. 

178 The red-slot indices (16, 52, 88) always decode as red fives. 

179 

180 Args: 

181 indices: 136-tile integers in the range 0-135, each naming a distinct 

182 physical tile. 

183 

184 Returns: 

185 The tiles named, in order. 

186 

187 Raises: 

188 NotationError: If an index is not an integer in 0-135, or a physical 

189 tile is named more than once. 

190 """ 

191 tiles: list[Tile] = [] 

192 seen: set[int] = set() 

193 for position, value in enumerate(indices): 

194 if not isinstance(value, int) or not 0 <= value < _INDEX_COUNT: 

195 raise NotationError(f"136-tile index must be an integer in 0-135, got {value!r} at position {position}") 

196 if value in seen: 

197 raise NotationError(f"duplicate 136-tile index {value} at position {position}") 

198 seen.add(value) 

199 tiles.append(Tile(TileKind(value // _SLOTS_PER_KIND), red=value in _RED_INDICES)) 

200 return tiles

203def dump_136(tiles: Iterable[Tile]) -> list[int]: 

204 """Serialize tiles to distinct 136-tile integers, in the given order. 

205 

206 Each tile takes the lowest free slot of its variant: a red five its one 

207 reserved slot, an ordinary five the three slots above it, every other kind 

208 its four. A collection with more copies of a variant than it has slots 

209 (four ordinary fives of one suit) is rejected. 

210 

211 Args: 

212 tiles: The tiles to serialize; their order is preserved. 

213 

214 Returns: 

215 Distinct 136-tile integers, one per tile. 

216 

217 Raises: 

218 NotationError: If a variant appears more times than it has physical 

219 copies (e.g. four ordinary fives of one suit). 

220 """ 

221 used: Counter[tuple[TileKind, bool]] = Counter() 

222 indices: list[int] = [] 

223 for tile in tiles: 

224 base = tile.kind * _SLOTS_PER_KIND 

225 if tile.red: 

226 first, capacity = base, 1 

227 elif tile.kind in FIVE_KINDS: 

228 first, capacity = base + 1, _SLOTS_PER_KIND - 1 

229 else: 

230 first, capacity = base, _SLOTS_PER_KIND 

231 slot = used[tile.kind, tile.red] 

232 if slot >= capacity: 

233 raise NotationError(f"no 136-tile slot left for {dump_mpsz([tile])!r} (already used {capacity})") 

234 used[tile.kind, tile.red] += 1 

235 indices.append(first + slot) 

236 return indices
