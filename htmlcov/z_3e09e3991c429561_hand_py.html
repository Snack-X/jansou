15#: A hand's four group slots. 

16MAX_MELDS = 4

18#: Concealed tiles of a meldless hand at rest. 

19FULL_HAND_SIZE = 13

21_TILES_PER_MELD_SLOT = 3 

22_MELD_SIZE = {"CHII": 3, "PON": 3, "DAIMINKAN": 4, "ANKAN": 4, "SHOUMINKAN": 4}

50@dataclass(frozen=True) 

51class Meld: 

52 """A group set aside from the concealed part. 

53 

54 ``called`` is the claimed tile for a chii, pon, open kan, or added kan 

55 (``None`` for a closed kan); ``source`` is the opponent it came from; 

56 ``added`` is the fourth tile of an added kan. Well-formedness is enforced 

57 at construction. 

58 

59 Attributes: 

60 type: Which of the five meld kinds this is. 

61 tiles: The meld's tiles. 

62 called: The claimed tile among ``tiles``, or ``None`` for a closed kan. 

63 source: The opponent the claimed tile came from, or ``None`` for a 

64 closed kan. 

65 added: The tile added when upgrading a pon to a kan, set only for an 

66 added kan. 

67 """ 

68 

69 type: MeldType 

70 tiles: tuple[Tile, ...] 

71 called: Tile | None = None 

72 source: CallSource | None = None 

73 added: Tile | None = None 

74 

75 def __post_init__(self) -> None: 

76 """Normalize ``tiles`` to a tuple and enforce meld well-formedness. 

77 

78 Raises: 

79 InvalidHandError: If the tile count, composition, or claim data do 

80 not match the meld kind. 

81 """ 

82 object.__setattr__(self, "tiles", tuple(self.tiles)) 

83 expected = _MELD_SIZE[self.type.name] 

84 if len(self.tiles) != expected: 

85 raise InvalidHandError(f"{self.type.name} must have {expected} tiles, got {len(self.tiles)}") 

86 if self.type is MeldType.CHII: 

87 self._check_chii_tiles() 

88 elif len({tile.kind for tile in self.tiles}) != 1: 

89 raise InvalidHandError(f"{self.type.name} tiles must be identical in kind") 

90 self._check_claim() 

91 

92 def _check_chii_tiles(self) -> None: 

93 """A chii is three consecutive tiles of one suit.""" 

94 suits = {tile.suit for tile in self.tiles} 

95 if None in suits or len(suits) != 1: 

96 raise InvalidHandError("chii tiles must be number tiles of one suit") 

97 ranks = sorted(tile.rank for tile in self.tiles) # type: ignore[type-var] 

98 if ranks != list(range(ranks[0], ranks[0] + _RUN_LENGTH)): 

99 raise InvalidHandError(f"chii tiles must be consecutive ranks, got {ranks}") 

100 

101 def _check_claim(self) -> None: 

102 """The claimed tile, source, and added tile match the meld kind.""" 

103 if self.type is MeldType.ANKAN: 

104 if self.called is not None or self.source is not None or self.added is not None: 

105 raise InvalidHandError("a closed kan has no claimed tile, no source, and no added tile") 

106 return 

107 if self.called is None or self.called not in self.tiles: 

108 raise InvalidHandError(f"{self.type.name} must distinguish a claimed tile among its tiles") 

109 if self.source is None: 

110 raise InvalidHandError(f"{self.type.name} must record the opponent the claimed tile came from") 

111 if self.type is MeldType.CHII and self.source is not CallSource.KAMICHA: 

112 raise InvalidHandError("a chii may come only from kamicha") 

113 if self.type is MeldType.SHOUMINKAN: 

114 if self.added is None or self.added not in self.tiles: 

115 raise InvalidHandError("an added kan must distinguish the added tile among its tiles") 

116 elif self.added is not None: 

117 raise InvalidHandError(f"{self.type.name} has no added tile") 

118 

119 @property 

120 def is_open(self) -> bool: 

121 """Whether the meld breaks concealment: every kind but the closed kan.""" 

122 return self.type is not MeldType.ANKAN 

123 

124 @property 

125 def is_kan(self) -> bool: 

126 """Whether the meld is a quad of any kind.""" 

127 return self.type in (MeldType.DAIMINKAN, MeldType.ANKAN, MeldType.SHOUMINKAN)

130@dataclass(frozen=True) 

131class Hand: 

132 """One player's tiles: the concealed part plus melds in the order made. 

133 

134 Attributes: 

135 concealed: The concealed tiles, including any tile just drawn. 

136 melds: The melds, in the order they were made. 

137 """ 

138 

139 concealed: tuple[Tile, ...] 

140 melds: tuple[Meld, ...] = () 

141 

142 def __post_init__(self) -> None: 

143 """Normalize ``concealed`` and ``melds`` to tuples.""" 

144 object.__setattr__(self, "concealed", tuple(self.concealed)) 

145 object.__setattr__(self, "melds", tuple(self.melds)) 

146 

147 @property 

148 def all_tiles(self) -> tuple[Tile, ...]: 

149 """Every tile of the hand: the concealed part and each meld's tiles.""" 

150 meld_tiles = tuple(tile for meld in self.melds for tile in meld.tiles) 

151 return self.concealed + meld_tiles 

152 

153 @property 

154 def is_concealed(self) -> bool: 

155 """Concealed (menzen): no open call; closed kans do not break this.""" 

156 return all(not meld.is_open for meld in self.melds) 

157 

158 @property 

159 def has_melds(self) -> bool: 

160 """The coarser notion: any meld at all, closed kans included.""" 

161 return bool(self.melds) 

162 

163 @property 

164 def rest_size(self) -> int: 

165 """The concealed-tile count this hand has at rest, given its melds.""" 

166 return FULL_HAND_SIZE - _TILES_PER_MELD_SLOT * len(self.melds)

168 def validate(self, rules: Rules | None = None) -> None: 

169 """Check the structural hand constraints. 

170 

171 Validity is structural only: completeness, readiness, and positional 

172 legality are separate concerns. Without ``rules``, one red five per 

173 suit is allowed (the 136-tile arrangement); with ``rules``, the 

174 red-five flag decides. 

175 

176 Args: 

177 rules: The rules whose ``aka_dora`` flag governs how many red fives 

178 are allowed, or ``None`` to allow one red five per suit. 

179 

180 Raises: 

181 InvalidHandError: If the meld count, concealed-tile count, per-kind 

182 copies, or red-five count violate the structural constraints. 

183 """ 

184 if len(self.melds) > MAX_MELDS: 

185 raise InvalidHandError(f"a hand has at most {MAX_MELDS} melds, got {len(self.melds)}") 

186 if len(self.concealed) not in (self.rest_size, self.rest_size + 1): 

187 raise InvalidHandError( 

188 f"with {len(self.melds)} melds the concealed part must hold " 

189 f"{self.rest_size} or {self.rest_size + 1} tiles, got {len(self.concealed)}" 

190 ) 

191 kind_counts = Counter(tile.kind for tile in self.all_tiles) 

192 for kind, count in kind_counts.items(): 

193 if count > TILES_PER_KIND: 

194 raise InvalidHandError(f"kind {kind.name} appears {count} times; at most {TILES_PER_KIND} allowed") 

195 reds_allowed = 1 if rules is None or rules.aka_dora else 0 

196 red_counts = Counter(tile.kind for tile in self.all_tiles if tile.red) 

197 for kind, count in red_counts.items(): 

198 if count > reds_allowed: 

199 raise InvalidHandError(f"{count} red fives of {kind.name}; at most {reds_allowed} allowed") 

200 

201 def is_valid(self, rules: Rules | None = None) -> bool: 

202 """Whether ``validate`` passes. 

203 

204 Args: 

205 rules: The rules passed through to ``validate``. 

206 

207 Returns: 

208 ``True`` if the hand satisfies the structural constraints. 

209 """ 

210 try: 

211 self.validate(rules) 

212 except InvalidHandError: 

213 return False 

214 return True
